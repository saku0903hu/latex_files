\section{微分方程式}
\begin{align}
  \frac{d}{dt} x &= \alpha x +\beta\\
  &= \alpha (x +\frac{\beta}{\alpha})&\\
  \frac{d}{dt} (x+\frac{\beta}{\alpha}) = \alpha (x +\frac{\beta}{\alpha})
\end{align}
とすれば線形な微分方程式に帰着する。
